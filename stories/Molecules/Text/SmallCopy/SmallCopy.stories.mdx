export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {locationlabel:'USA', location: 'UNDP/Angola', phonelabel:'Phone', phone: '+1 212-906-5000', emaillabel:'Email', email: 'hq@undp.org'}; return engText;
    case 'arabic': const arabicText = {locationlabel:'الولايات المتحدة الأمريكية', location: 'برنامج الأمم المتحدة الإنمائي / أنغولا', phonelabel:'هاتف', phone: '+1 212-906-5000', emaillabel:'البريد الإلكتروني', email: 'hq@undp.org'}; return arabicText;
    case 'burmese': const burmeseText = {locationlabel:'ယူအက်စ်အေ', location: 'UNDP/အန်ဂိုလာ', phonelabel:'ဖုန်း', phone: '+1 212-906-5000', emaillabel:'အီးမေးလ်', email: 'hq@undp.org'}; return burmeseText;
    case 'japanese': const japaneseText = {locationlabel:'宇佐', location: 'UNDP /アンゴラ', phonelabel:'電話', phone: '+1 212-906-5000', emaillabel:'Eメール', email: 'hq@undp.org'}; return japaneseText;
    default:
      return {locationlabel:'USA', location: 'UNDP/Angola', phonelabel:'Phone', phone: '+1 212-906-5000', emaillabel:'Email', email: 'hq@undp.org'};
  }
};

# Small Copy Lockup Component
Small Copy Lockup component defines a section of information data with some items - Location, Phone Number and Email ID.

<Story name="Small Copy Lockup">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <Smallcopy locationName={caption.locationlabel} locationText={caption.location} phoneName={caption.phonelabel} phonenumber={caption.phone} emailName={caption.emaillabel} emailid={caption.email}></Smallcopy>;
  }}
</Story>
