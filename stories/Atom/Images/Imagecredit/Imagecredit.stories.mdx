export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'UNDP/Angola', name: 'Jerome Bell'}; return engText;
    case 'arabic': const arabicText = {detail: 'برنامج الأمم المتحدة الإنمائي / أنغولا', name: 'جيروم بيل'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'UNDP/အန်ဂိုလာ', name: 'ဂျရုမ်း ဘဲလ်'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'UNDP /アンゴラ', name: 'ジェロームベル'}; return japaneseText;
    default:
      return {detail: 'UNDP/Angola', name: 'Jerome Bell'};
  }
};

# Image Credit Component
Image Credit component used to show the image caption and who posted the image in a related image.

<Story name="Imagecredit">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <Imagecredit label={caption.detail} name={caption.name}></Imagecredit>;
  }}
</Story>
