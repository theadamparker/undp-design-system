export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Label'}; return engText;
    case 'arabic': const arabicText = {detail: 'ملصق'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'တံဆိပ်'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'ラベル'}; return japaneseText;
    default:
      return {detail: 'Label'};
  }
};
